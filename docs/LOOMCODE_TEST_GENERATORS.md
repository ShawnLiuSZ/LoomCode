## 方式一：go:generate + text/template

适合 Provider、Tool 等接口数量有限的场景：

```go
// internal/testutil/gen.go
//go:generate go run gen_provider.go

package testutil
```

```go
// internal/testutil/gen_provider.go
//go:build ignore

package main

import (
    "os"
    "text/template"
)

const mockProviderTemplate = `// Code generated by gen_provider.go; DO NOT EDIT.
package testutil

import (
    "context"
    "errors"
    "sync"
)

type MockProvider struct {
    mu sync.Mutex
    ChatFunc    func(ctx context.Context, req *ChatRequest) (*ChatResponse, error)
    chatCalls   []chatCall
    StreamFunc  func(ctx context.Context, req *ChatRequest) (<-chan StreamEvent, error)
    streamCalls []streamCall
    ModelsFunc  func() []ModelInfo
    CapsFunc    func() Capabilities
    NameFunc    func() string
}

type chatCall struct {
    Ctx context.Context
    Req *ChatRequest
}

func (m *MockProvider) Chat(ctx context.Context, req *ChatRequest) (*ChatResponse, error) {
    m.mu.Lock()
    m.chatCalls = append(m.chatCalls, chatCall{Ctx: ctx, Req: req})
    m.mu.Unlock()
    if m.ChatFunc == nil {
        return nil, errors.New("ChatFunc not set")
    }
    return m.ChatFunc(ctx, req)
}

func (m *MockProvider) ChatCalled() []chatCall {
    m.mu.Lock()
    defer m.mu.Unlock()
    return append([]chatCall{}, m.chatCalls...)
}

func (m *MockProvider) ChatCallCount() int {
    m.mu.Lock()
    defer m.mu.Unlock()
    return len(m.chatCalls)
}
// ... 其他方法类似
`

func main() {
    tmpl := template.Must(template.New("mock").Parse(mockProviderTemplate))
    f, _ := os.Create("mock_provider.go")
    defer f.Close()
    tmpl.Execute(f, nil)
}
```

运行 `go generate ./internal/testutil/` 生成 `mock_provider.go`。

## 方式二：接口契约驱动

在接口定义处用注释声明 mock 生成规则：

```go
// internal/provider/provider.go

//go:generate mockgen -source=provider.go -destination=../testutil/mock_provider.go -package=testutil

type Provider interface {
    Chat(ctx context.Context, req *ChatRequest) (*ChatResponse, error)
    Stream(ctx context.Context, req *ChatRequest) (<-chan StreamEvent, error)
    Models() []ModelInfo
    Capabilities() Capabilities
    Name() string
}
```

Phase 1 阶段优先手写以保持零依赖；Phase 2+ 接口稳定后可引入 `github.com/golang/mock`。

## 集成示例 1：Provider → MockProvider

**接口定义** (`internal/provider/provider.go`)：

```go
package provider

//go:generate go run ../../internal/testutil/gen_provider.go

type Provider interface {
    Chat(ctx context.Context, req *ChatRequest) (*ChatResponse, error)
    Stream(ctx context.Context, req *ChatRequest) (<-chan StreamEvent, error)
    Models() []ModelInfo
    Capabilities() Capabilities
    Name() string
}
```

**生成器** (`internal/testutil/gen_provider.go`) 用 `methodSpec` 表驱动：

```go
type methodSpec struct {
    Name       string
    Params     string
    Returns    string
    FuncType   string
    CallType   string
    CallFields string
    ZeroReturn string
    ArgAssign  string
}

var providerMethods = []methodSpec{
    {
        Name: "Chat", Params: "ctx context.Context, req *ChatRequest",
        Returns: "(*ChatResponse, error)",
        CallType: "chatCall", CallFields: "Ctx context.Context; Req *ChatRequest",
        ZeroReturn: "nil, errors.New(\"ChatFunc not set\")",
        ArgAssign: "Ctx: ctx, Req: req",
    },
    // ... Stream, Models, Capabilities, Name
}
```

生成的 `mock_provider.go` 每个方法包含：`XxxFunc`（可注入实现）、`xxxCalls`（调用记录）、`XxxCalled()`（获取记录）、`XxxCallCount()`（获取次数）。

## 集成示例 3：AgentLoop 综合测试

```go
func TestAgentLoop_FullIntegration(t *testing.T) {
    // 1. 生成的 MockProvider
    mockProvider := &testutil.MockProvider{}
    mockProvider.ChatFunc = func(ctx context.Context, req *ChatRequest) (*ChatResponse, error) {
        if mockProvider.ChatCallCount() == 1 {
            return &ChatResponse{
                ToolCalls: []ToolCall{{ID: "call_1", Name: "read_file", Args: map[string]any{"path": "main.go"}}},
            }, nil
        }
        return &ChatResponse{Content: "代码分析完成"}, nil
    }

    // 2. 表驱动 StubToolRegistry
    tools := testutil.NewStubToolRegistry()

    // 3. callRecorder 验证守卫链
    recorder := &testutil.CallRecorder{}

    // 4. 组装
    agent := NewAgent(mockProvider, tools)
    agent.AddGuard(func(tc ToolCall) error {
        recorder.Record("guard", tc.Name)
        return nil
    })

    // 5. 执行 + 断言
    result, err := agent.Run(context.Background(), "分析 main.go")
    if err != nil {
        t.Fatal(err)
    }
    if result != "代码分析完成" {
        t.Errorf("unexpected result: %s", result)
    }
    if mockProvider.ChatCallCount() != 2 {
        t.Errorf("expected 2 Chat calls, got %d", mockProvider.ChatCallCount())
    }
    if recorder.Count() < 1 {
        t.Error("guard was not invoked")
    }
}
```
